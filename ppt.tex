\section{components}
\begin{frame}[fragile]
\frametitle{Components of a LaTeX Document}
Every LaTeX document must contain the following three components. Everything else is optional (even text).\\
\begin{itemize}
 \item \textbackslash{}documentclass\{beamer\}
  document class may be Article, Report, Book, IEEEtran, Letter \\
\item \textbackslash{}begin\{document\} ...\textbackslash{}end\{document\}
\end{itemize}
\begin{example}
\small
\begin{verbatim}
\documentclass{article}
\begin{document}
Hello world!
\end{document}
\end{verbatim}
\end{example}
\end{frame}



%----------fifth slide----------
\section{principles}
\begin{frame}
\frametitle{Principles}
\begin{itemize}
\item Commands are case sensitive
\item LaTeX commands in braces {}
\item Optional information is included in square brackets [].
\item The \textbackslash{}documentclass\ command must appear at the very beginning of your LaTeX document, before any other LaTeX commands, or you will get an error message.
\item Commands that affect the whole document, should be included in the preamble,  space between the \textbackslash{}documentclass\ and \textbackslash{}begin\{document\} commands is called.
\end{itemize}
\end{frame}
